Save American Hemp Countdown Widget - Demo Page

1. Default Gray Theme

<div id="sah-countdown"></div>
<script src="https://savehempclock.com/sah-countdown-widget.js"></script>
<script>
  SAHCountdown.init({
    container: '#sah-countdown'
  });
</script>

2. Dark Theme

<script>
  SAHCountdown.init({
    container: '#sah-countdown',
    theme: 'dark'
  });
</script>

3. Light Theme

<script>
  SAHCountdown.init({
    container: '#sah-countdown',
    theme: 'light'
  });
</script>

4. Custom Colors (Green Theme)

<script>
  SAHCountdown.init({
    container: '#sah-countdown',
    colors: {
      panelBackground: '#006400',
      panelText: '#ffffff',
      title: '#228B22',
      hashtag: '#228B22',
      containerBackground: '#f0fff0'
    }
  });
</script>

5. Compact Size

<script>
  SAHCountdown.init({
    container: '#sah-countdown',
    size: 'compact'
  });
</script>

6. Clock Only (Minimal)

<script>
  SAHCountdown.init({
    container: '#sah-countdown',
    showTitle: false,
    showSubtitle: false,
    showDescription: false,
    showHashtag: false
  });
</script>

7. With CTA Button

<script>
  SAHCountdown.init({
    container: '#sah-countdown',
    showCTA: true,
    showDescription: false
  });
</script>
